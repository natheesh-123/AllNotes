=> c# code to insert iamge


using MySql.Data.MySqlClient;
using System;
using System.IO;

class Program
{
    static void Main()
    {
        string connectionString = "server=localhost;user=root;database=mydatabase;port=3306;password=mypassword";

        using (MySqlConnection conn = new MySqlConnection(connectionString))
        {
            conn.Open();

            // Prepare the SQL command
            string sql = "INSERT INTO Users (name, email, password, role, profile_picture) VALUES (@name, @email, @password, @role, @profile_picture)";

            using (MySqlCommand cmd = new MySqlCommand(sql, conn))
            {
                // Set the parameters
                cmd.Parameters.AddWithValue("@name", "John Doe");
                cmd.Parameters.AddWithValue("@email", "john.doe@example.com");
                cmd.Parameters.AddWithValue("@password", "hashed_password");
                cmd.Parameters.AddWithValue("@role", "user");

                // Load the image
                byte[] imageData = File.ReadAllBytes("C:\\path\\to\\your\\image.jpg");
                cmd.Parameters.AddWithValue("@profile_picture", imageData);

                // Execute the command
                cmd.ExecuteNonQuery();
            }
        }
    }
}






note:
Make sure that the MySQL server has permission 
to read the file and that the secure_file_priv system variable is 
configured correctly (pointing to the directory where the image is stored).
